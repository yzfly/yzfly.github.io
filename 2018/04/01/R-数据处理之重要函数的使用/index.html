lapply

可以循环处理列表中的每一个元素
lapply(参数)： lapply(列表，函数/函数名,其他参数)
lapply总是返回一个列表
str(apply) #str()把任意的R对象以一种整洁紧凑的方式展现出来

x <- list(a=1:10,b=c(11,21,34,53,54)) #创建一个列表
lapply(x,mean)                #将列表中的两个元素分布求平均

x <- 1:4
lapply(x,runif)   #利用lapply将runif函数依次作用于x中的每一个元素
lapply(x,runif,min=0,max=100) #min 和max对runif函数产生的值的范围进行了限制
lapply(x,function(m) m[1,]) #编写自己的函数进行循环
#此处函数为 function(m) m[1,] ,传入矩阵m，取出m中的第一行

sapply

sapply:简化结果
当结果列表元素的长度均为一的时候，返回向量
当结果列表元素长度相同且大于1的时候，返回矩阵
》结果列表元素长度不同时，与lapply的结果相同

apply

沿着数组的某一维度处理数据
apply(数组,维度,函数/函数名)

x <- matrix(1:16,4,4)
apply(x,2,mean)        #对矩阵的第二个维度（列）求平均
apply(x,2,sum)  #对每一列进行求和
apply(x,2,mean)  #对每一列求平均

rowSums(x)    #求行和
rowMeans(x)    #求行平均
colMeans(x)
colSums(x)    #求列和

mapply

lapply的多元版本
mapply(函数/函数名,数据，函数相关的参数)

x <- list(rep(1,4),rep(2,3),rep(3,2),rep(4,1))
y <- mapply(rep,1:4,4:1)   #x列表 和y列表相同

s <- function(n,mean,std)｛    #()内为传入函数中的变量，花括号内为函数体
    rnorm(n,mean,std)
｝
s(4,0,1)                #运行函数

mapply(s,1:5,5:1,2)  # mapply依次将参数传入s函数中进行处理
list(s(1,5,2),s(2,4,2),s(3,3,2),s(4,2,2),s(5,1,2))

排序

sort 对向量进行排序，返回拍好序的内容
order 返回拍好序的内容的下标，可以按照多个标准进行排序
x <- data.frame(v1=1:5,v2=c(10,7,9,6,8),v3=11:15,v4=c(1,1,2,2,1))
sort(x$v2) #对数据框中的v2升序排列
sort(x$v2,decreasing=TRUE) #对数据框中的v2降序排列

order(x$v2)   #order函数返回的是排序数据的行号
x[order(x$v2),]         #对x按照v2排序
x[order(x$v4,x$v2),]        #按照v4主排序，v4值相同时按照v2排序
x[order(x$v2,x$v4,decreasing=TRUE),]   #降序排序
