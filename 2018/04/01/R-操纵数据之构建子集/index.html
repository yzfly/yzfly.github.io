R的学习笔记3

R处理数据过程中，拿到的为原始数据（raw dataset），需要对数据进行预处理得到预处理之后的数据（clean dataset）,这就涉及到提取元素构建子集

构建子集的基本方法

[] 提取一个或多个类型相同的元素
[[]] 从列表或者数据框中提取元素
$ 按名字从列表或数据框中提取元素
x <- 1:10
x[1] #取出第一个元素
x[4:7] #取出第4到7的元素

x[x>5]     #取出满足一定条件的数据
x[ x>3 & x<8 ]
x[x<3 | x>7]

x <- c(1,2,3,4)   #按照元素名获取元素值
names(x) <- c("a","b","c","d")
x["b"]

矩阵的子集

x <- matrix(1:6,nrow=2,ncol=3)
x[1,2]        #获取第一行第二列的元素
x[2,3]    #获取第二行第三列的元素
x[1,]    #获取第1行的所有数据
x[,2]    #获取第2列的所有数据
x[2,c(1,3)]    #获取第2行的1，3列的所有数据

x[1,2,drop=FALSE]  #使返回值为矩阵而不是默认的向量

数据框的子集

x <- data.frame(v1=1:5,v2=6:10,v3=11:15)   #生成数据框

x[2,]  # 类似矩阵的操作是可行的
x[,3]
x[1,4]

x[,"v2"]  #使用列名提取整列

x$v2[c(2,4)] #提取v2列中的2，4行元素

x[(x$v1<4 & x$v2>=8),]  #使用逻辑判断提取特定值
# 提取  v1列中值小于4并且v2列中值大于8的所有行，每一列都保留

which(x$v1>2)  #判断命令，返回下标
x$v1>2        #直接判断语句，返回逻辑值

subset(x,x$v1>2)  #构建满足条件的数据框

列表的子集

子集获取方法大体为所列的几种： [[]] $ [[]][] [[]][[]]

l <- list(id=c(1,2,3,4),a=list(1,2,3,4),gender=c("male","female"))

l[1]  #获取列表中的某个元素，包含元素名和元素值
l['id']

l[[1]] #获取列表中的某个元素，只获取其内容
l[["id"]]
l$id

l[c(1,3)] #获取列表中多个元素

#获取嵌套元素的值（此处为列表中的列表）
l[[c(2,3)]]   #获取l列表中第二个元素（a列表）中的第二个元素的值
l[[c(3,1)]]      #获取l列表中第三个元素中的第一个元素的值

l[[2]][[2]]      #获取l列表中第二个元素（a列表）中第二个元素的值
l[[2]][2]    #获取l列表中第二个元素（a列表）中第二个元素（包含属性和值）

不完全匹配

l2 <- list(amalf = 1:10)

l2$amalf   #取出元素（包含元素和值）
l2$a        #不输入全名也可以达到同样的效果

l2[["a"]]  #此处为不完全匹配的错误示范
l2[["a",exact=FALSE]] #不完全匹配的正确示范

缺失值处理

x <- c(1,NA,2,3,NA)
is.na(x)   #判断缺失值
x[is.na(x)]  #获取向量中的缺失值
x[!is.na(x)] #获取向量中的非缺失值

x <- c(1,NA,2,NA,3)        #获取x y中对应位置都不是缺失值的值
y <- c("a","b",NA,"c",NA)
z <- complete.cases(x,y)  #x y对应位置没有缺失值时该位置为TRUE
x[z]   #获取对应位置值
y[z]

# 处理数据集中的缺失值
library(datasets)   #导入数据集包，里面内置了许多数据集
head(airquality)    #查看airguality数据集头部（看一下数据集是怎样的）
g <- complete.cases(airquality)
airquality[g,][1:10,] #获取没有缺失值的那些行中的前十行，保留所有列

向量化操作

向量化操作是指对向量进行统一操作的过程
x <- 1:5
y <- 6:10
x + y #向量对应元素相加
x*y #向量对应元素相乘
x/y #向量对应元素相除

m <- matrix(1:4,nrow=2,ncol=2)        #创建两个矩阵
n <- matrix(rep(2,4),nrow=2,ncol=2)  # rep(2,4)表示把2重复四次
m*n   #矩阵对应元素相乘
m %*% n # 线性代数矩阵乘法
