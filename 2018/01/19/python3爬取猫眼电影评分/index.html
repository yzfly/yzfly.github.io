猫眼电影评分获取
发布于：2018-01-19 23:09:32 访问：

python版本： python3.6.4 64bit

爬取猫眼电影上的文章评分，个人学习记录，非学习教程~

python爬取流程： 网页抓取器-> 网页解析器 -> 文件保存器
关键在于正则表达式的构建

在windows下会出现麻烦的编码问题，最好在linux上运行

代码如下： 

# python3
# _*_ coding:utf-8 _*_

import re  # 导入正则表达式库
import json  # 导入json库
from multiprocessing import Pool  # 导入python内置的多线程
import requests  # 导入request库获取网页信息
from requests.exceptions import RequestException


headers = {"User-agent": "Mozilla/5.0"}  # 浏览器头部信息


def get_one_page(url):    # 获取一个页面
    try:
        res = requests.get(url, headers=headers)
        if res.status_code == 200:
            return res.text
        return None
    except RequestException:
        return None


def parse_one_page(html):  # 解析一个页面

    pattern = re.compile('<dd>.*?board-index.*?>(\d+)</i>.*?data-src="(.*?)".*?name"><a' +
                         '.*?>(.*?)</a>.*?star">(.*?)</p>.*?releasetime">(.*?)</p>' + '.*?integer">(.*?)</i>.*?fraction">(.*?)</i>.*?</dd>', re.S)
    # 正则表达式匹配
    items = re.findall(pattern, html)   # 正则表达式处理字符串，找到所有上一步中匹配的记录
    for item in items:
        yield{
            'index': item[0],            # 提取出列表中的信息
            'image': item[1],
            'title': item[2],
            'actor': item[3].strip()[3:],
            'time': item[4].strip()[5:],
            'score': item[5]+item[6]
        }


def write_to_file(content):             # 将解析器获取到的信息写入文件之中
    with open('result.txt', 'a', encoding='utf-8') as f:
        # 有中文字符，关闭ascii编码，加上换行符
        f.write(json.dumps(content, ensure_ascii=False)+'\n')
        f.close()    # 关闭


def main(offset):
    url = 'http://maoyan.com/board/4?offset='+str(offset)   # 构造榜单页面url

    html = get_one_page(url)    # 获取html页面
    for item in parse_one_page(html):  # 解析html页面的信息
        print(item)                    # 输出解析信息
        write_to_file(item)               # 信息写入文件

    # if __name__ == '__main__':         # 这段原理没搞清楚所以注释掉
    #     p = Pool(10)                     # 创建具有10个进程的进程池
    #     p.map(main, [ i*10 for i in range(10) ])   # 使用多线程爬取
    #     p.close()  # 关闭进程池，不再接受新的进程

for i in range(10):
    main(i*10)
