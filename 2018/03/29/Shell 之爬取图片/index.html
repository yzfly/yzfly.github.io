Shell 之爬取图片

系统： ubuntu16.04 64bit
Bash shell 之 自动下载数字图像处理配套图片资源并解压缩

自动下载各个章节的图片

#!/bin/bash
#This is a shell script for downloading the Images for a book
#Author:liushang
mkdir /root/Images
cd /root/Images
yum install -y zip unzip
apt-get install -y zip unzip      #Install software
for((i=1;i<=12;i++));
do   
    # download zip files
    if [ $i -lt 10 ]
        then wget "http://www.imageprocessingplace.com/downloads_V3/dip3e_downloads/dip3e_book_images/DIP3E_CH0`echo $i`_Original_Images.zip"
        unzip "DIP3E_CH0`echo $i`_Original_Images.zip"
    else
        wget "http://www.imageprocessingplace.com/downloads_V3/dip3e_downloads/dip3e_book_images/DIP3E_CH`echo $i`_Original_Images.zip"
        unzip "DIP3E_CH`echo $i`_Original_Images.zip"
    fi
done
cd /root/Images && rm -rf DIP*.zip         #删除zip压缩文件
cd /root/ && zip -r Images.zip Images   #压缩下载下来的图片的文件

echo "All Images have been downloaded~ enjoy~"
